"""
Pydantic schema for pending translation approval (append to existing translation.py)
"""
# Add this to the end of the existing translation.py file

class PendingTranslationResponse(BaseModel):
    """Detailed response for pending translations"""
    id: UUID
    part_name_en: str
    part_name_pr: Optional[str] = None
    part_name_fr: Optional[str] = None
    hs_code: Optional[str] = None
    category_en: Optional[str] = None
    approval_status: str
    submitted_at: Optional[datetime] = None
    reviewed_at: Optional[datetime] = None
    rejection_reason: Optional[str] = None
    created_by: Optional[UUID] = None
    created_at: datetime
    
    class Config:
        from_attributes = True
